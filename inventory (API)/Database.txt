CREATE DATABASE inventory;
USE inventory;

-- Tabel User untuk Login/Register
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL
);

-- Tabel Barang
CREATE TABLE items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nama_barang VARCHAR(100),
    stok INT,
    harga DECIMAL(10,2)
);
